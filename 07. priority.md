```
#!/bin/bash

echo "Enter the number of processes:"
read n

# Arrays
arrival=()
burst=()
priority=()
process=()
ct=()
tat=()
wt=()
completed=()

# Input arrival, burst, priority
for ((i=0; i<n; i++))
do
    echo "Enter arrival time for P$((i+1)):"
    read at
    arrival[$i]=$at

    echo "Enter burst time for P$((i+1)):"
    read bt
    burst[$i]=$bt

    echo "Enter priority for P$((i+1)) (lower number = higher priority):"
    read pr
    priority[$i]=$pr

    process[$i]=$((i+1))
    completed[$i]=0
done

time=0
count=0

# Main loop - Priority Scheduling (Non-preemptive)
while [ $count -lt $n ]
do
    min_pr=9999
    min_index=-1

    # Find highest priority among arrived & not completed
    for ((i=0; i<n; i++))
    do
        if [ ${arrival[$i]} -le $time ] && [ ${completed[$i]} -eq 0 ]
        then
            if [ ${priority[$i]} -lt $min_pr ]
            then
                min_pr=${priority[$i]}
                min_index=$i
            fi
        fi
    done

    if [ $min_index -eq -1 ]
    then
        ((time++))     # no process arrived yet
    else
        ct[$min_index]=$((time + burst[$min_index]))
        tat[$min_index]=$(( ct[$min_index] - arrival[$min_index] ))
        wt[$min_index]=$(( tat[$min_index] - burst[$min_index] ))

        time=${ct[$min_index]}
        completed[$min_index]=1
        ((count++))
    fi
done

# Display results
echo ""
echo "Process  Arrival  Burst  Priority  Completion  TurnAround  Waiting"
for ((i=0; i<n; i++))
do
    echo "P${process[$i]}        ${arrival[$i]}        ${burst[$i]}        ${priority[$i]}          ${ct[$i]}          ${tat[$i]}          ${wt[$i]}"
done

# Calculate averages
total_wt=0
total_tat=0

for ((i=0; i<n; i++))
do
    total_wt=$(( total_wt + wt[$i] ))
    total_tat=$(( total_tat + tat[$i] ))
done

avg_wt=$(echo "scale=2; $total_wt / $n" | bc)
avg_tat=$(echo "scale=2; $total_tat / $n" | bc)

echo ""
echo "Average Waiting Time: $avg_wt"
echo "Average Turnaround Time: $avg_tat"

# Simple Gantt chart (same style)
echo ""
echo "Gantt Chart:"
echo -n "|"
for ((i=0; i<n; i++))
do
    echo -n " P${process[$i]} |"
done

echo ""
printf "%-4s" "0"
for ((i=0; i<n; i++))
do
    printf "%-8s" "${ct[$i]}"
done
echo ""
```
